\newcommand\bccw{%
\mathop{\tikz[scale=0.24]{\draw[thick,-{Triangle[scale=0.5]}]
              (0,0)to[out=0,in=270](1,1);}}%
}
\newcommand\bcw{%
\mathop{\tikz[scale=0.24]{\draw[thick,-{Triangle[scale=0.5]}]
              (0,0)to[out=180,in=-90](-1,1);}}
}
\newcommand\tcw{%
\mathop{\tikz[scale=0.24]{\draw[thick,-{Triangle[scale=0.5]}]
              (0,0)to[out=90,in=180](1,1);}}
}
\newcommand\tccw{%
\mathop{\tikz[scale=0.24]{\draw[thick,-{Triangle[scale=0.5]}]
              (0,0)to[out=90,in=0](-1,1);}}
}
\newcommand\aup{%
\mathop{\tikz[scale=0.24]{\draw[thick,-{Triangle[scale=0.5]}]
              (0,0)to[out=90,in=270](0,1)}}
}
\newcommand\adown{%
\mathop{\tikz[scale=0.24]{\draw[thick,-{Triangle[scale=0.5]}]
              (0,1)to[out=90,in=90](0,0)}}
}
\newcommand\bccwdown{%
\mathop{\tikz[scale=0.24]{\draw[thick,-{Triangle[scale=0.5]}]
              (0,0)to[out=-90,in=180](1,-1);}}
}
\newcommand\bcwdown{%
\mathop{\tikz[scale=0.24]{\draw[thick,-{Triangle[scale=0.5]}]
              (0,0)to[out=180,in=90](-1,-1);}}
}
